(defun greet (name) (format "Hello, ~a!" name))
(defun multiply (x y) (* x y))
(multiply 4 5)
(defun doublen (n) (* 2 n))
(doublen 10)
(defun hello ()
  (format t "Hello Coding Challenge World~%"))
(hello)
(defun doublen2 (n)
  (* n 2))
(doublen2 100)
(defun fib (n)
  (if (< n 2)
      n
      (+ (fib (- n 1))
         (fib (- n 2)))))
(fib 20)
(defun fact (n)
  (if (<= n 1)
    1
    (* n (fact (- n 1)))))
(fact 10)
